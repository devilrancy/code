int fib(int m)
{
  int a,b,c;
  a=0;
  b=1;
  c=ab;
  while(c<m)
  {
  a=b;
  b=c;
  c=ab;
  }
  return b;
}
int rfibsearch(int a[],int n,int p,int q,int r,int key)
{
  int t;
  if(p<1||p>n)
  return 0;
  else if(key==a[p])

 {
  if(r==0)
  return 0;
  else
  {
  p=p-r;
  t=q;
  q=r;
  r=t-r;
  return rfibsearch(a,n,p,q,r,key);
  }
  }
  else
  {
  if(q==1)

  else
  {
  p=pr;
  q=q-r;
  r=r-q;

  }
  }
}
